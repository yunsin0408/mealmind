document.querySelectorAll('input[name="mode"]').forEach(radio => {
    radio.addEventListener('change', function() {
        const pantryDiv = document.getElementById('pantryInputs');
        const randomDiv = document.getElementById('randomInputs');
        if (this.value === 'pantry') {
            pantryDiv.style.display = 'block';
            randomDiv.style.display = 'none';
        } else {
            pantryDiv.style.display = 'none';
            randomDiv.style.display = 'block';
        }
    });
});
